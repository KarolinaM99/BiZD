Zadanie 1
SELECT (last_name||' '||salary) AS "wynagrodzenie" FROM EMPLOYEESWHERE department_id >= 20 AND department_id <= 50 AND salary >= 2000 AND salary <= 7000ORDER BY last_name;Zadanie 2SELECT hire_date, last_name, &podaj_nazwe_kolumny FROM EMPLOYEESWHERE hire_date >= '2005-01-01' AND hire_date <= '2005-12-31'ORDER BY &podaj_nazwe_kolumny;Zadanie 3SELECT (first_name||' '||last_name) as imie_nazwisko, salary, phone_number FROM EMPLOYEESWHERE last_name LIKE '__e%' AND first_name LIKE '%&podaj_czesc_imienia%'ORDER BY imie_nazwisko DESC, salary ASC;Zadanie 4SELECT first_name, last_name, MONTHS_BETWEEN(SYSDATE, hire_date) AS liczba_miesiecy,    CASE        WHEN MONTHS_BETWEEN(SYSDATE, hire_date) <= 150 THEN ROUND(0.1 * salary, 2)        WHEN MONTHS_BETWEEN(SYSDATE, hire_date) <= 200 THEN ROUND(0.2 * salary, 2)        ELSE ROUND(0.3 * salary, 2)    END AS wysokosc_dodadkuFROM EMPLOYEESORDER BY liczba_miesiecy;Zadanie 5SELECT JOBS.job_title, SUM(EMPLOYEES.salary), ROUND(AVG(EMPLOYEES.salary)) FROM EMPLOYEES JOIN JOBS ON EMPLOYEES.job_id = JOBS.job_idWHERE JOBS.min_salary > 5000GROUP BY JOBS.job_titleORDER BY JOBS.job_title;
